Part 2: Geomapping Tweet Sentiment

Sentiment Visualization

var width = 960, height = 750; var rateById = d3.map(); var rateByName=d3.map(); var quantize= d3.scale.threshold() .domain([0.01, 0.4, 0.428, 0.457, 0.486, 0.514, 0.542, 0.571, 0.64]) .range(d3.range(10).map(function(i) { return "q" + i + "-9"; })); var projection = d3.geo.albersUsa() //getsthe map .scale(1280) .translate([width / 2, height / 2]); var path = d3.geo.path() .projection(projection); var svg = d3.select("#gMap").append("svg") .attr("width", width) .attr("height", height); queue() .defer(d3.json, "us.json") .defer(d3.tsv, "county_sentiments.tsv", function(d) { rateById.set(d.id, +d.sent); rateByName.set(d.id, d.name);}) .await(ready); function ready(error, us) { svg.append("g") .attr("class", "counties") .selectAll("path") .data(topojson.feature(us, us.objects.counties).features) .enter() .append("path") .attr("class", function(d) { return quantize(rateById.get(d.id)); }) .attr("d", path) .append("title") //adds the hovering effect .text(function(d){ return rateByName.get(d.id) + "\t" + rateById.get(d.id).toFixed(2); }); svg.append("path") .datum(topojson.mesh(us, us.objects.states, function(a, b) { return a !== b; })) .attr("class", "states") .attr("d", path); }
